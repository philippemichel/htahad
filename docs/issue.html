hexptabph <-
  function(dfk,
           exp = FALSE,
           nomfich = "export",
           nomsheet = "x",
           lg = FALSE) {
    zz <- dfk |>
      as_kable_extra() |>
      kable_styling(bootstrap_options = c("striped", "hover"),
                    fixed_thead = TRUE) |> 
      scroll_box(width = "100%", height = "780px  ")
    if (exp) {
      dfk |>
        as_tibble() |>
        write_ods(path = nomfich, sheet = nomsheet, append = TRUE)
    }
    return(zz)
  }
